#include<stdio.h>

struct Student
{

	int stuId;
	int stuAge;
	char stuName[100];
	struct Student *next;
};
void printStudent(struct Student *curp)
{
	printf("学号: %d, 年龄 : %d , 姓名:%s\n",curp->stuId,curp->stuAge,curp->stuName);
}
void printLinkedList(struct Student *headp)
{
	while(headp!=NULL)
	{
		printStudent(headp);
		headp=headp->next;
	}
}
